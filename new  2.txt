		function collide(bullets, enermies) {
			for (var i = 0; i < bullets.length; i++) {
				for (var j = 0; j < enermies.length; j++) {
					x1 = bullets[i].getX();
					y1 = bullets.[i].getY();
					x2 = enermies[j].getX();
					y2 = enermies[j].getY();
					dx = Math.pow((x1 - x2), 2);
					dy = Math.pow((y1 - y2), 2);
					distance = Math.sqrt(dx + dy);
					if (distance <= 5) {
						enermies[j].remove();
					}
				}
			}
		}
